FCFS (First-Come, First-Served) Scheduling Algorithm

The following C program implements the FCFS scheduling algorithm.

#include<stdio.h>
#include<conio.h>
int main()
{
    int n,bt[10],wt[10],tat[10],i,j;
    float awt=0,atat=0;
    printf("\nenter the no.of process:");
    scanf("%d",&n);
    printf("enter process burst time:\n");
    for(i=0;i<n;i++)
    {
        printf("p[%d]:",i+1);
        scanf("%d",&bt[i]);
    }
    printf("\npid\tbt\twt\ttat");
    for(i=0;i<n;i++)
    {
        wt[i]=0;
        tat[i]=0;
        for(j=0;j<i;j++)
        {
            wt[i]=wt[i]+bt[j];
        }
        tat[i]=bt[i]+wt[i];
        awt=awt+wt[i];
        atat=atat+tat[i];
        printf("\n%d\t%d\t%d\t%d",i+1,bt[i],wt[i],tat[i]);
    }
    awt=awt/n;
    atat=atat/n;
    printf("\naverage wt:%f",awt);
    printf("\naverage tat:%f",atat);
    return 0;
}

SJF Scheduling Algorithm

The following C program implements the SJF scheduling algorithm.

#include<stdio.h>
#include<conio.h>
int main()
{
    int n,bt[10],wt[10],tat[10],ct[10],p[10],sum,i,j,k,temp;
    float totaltat=0,totalwt=0;
    printf("\nenter the no.of process:");
    scanf("%d",&n);
    printf("\nenter the process number:");
    for(i=0;i<n;i++)
        scanf("%d",&p[i]);
    printf("\nenter the burst time:");
    for(i=0;i<n;i++)
    {
        scanf("%d",&bt[i]);
    }
    for (i=0;i<n;i++)
    {
        for(j=0;j<n-i-1;j++)
        {
            if(bt[j]>bt[j+1])
            {
                temp=bt[j];
                bt[j]=bt[j+1];
                bt[j+1]=temp;
                temp=p[j];
                p[j]=p[j+1];
                p[j+1]=temp;
            }
        }
    }
    sum=0;
    for(j=0;j<n;j++)
    {
        sum=sum+bt[j];
        ct[j]=sum;
        printf("%d\t",ct[j]);
    }
    for(k=0;k<n;k++)
    {
        tat[k]=ct[k];
        totaltat=totaltat+tat[k];
    }
    for(k=0;k<n;k++)
    {
        wt[k]=tat[k]-bt[k];
        totalwt=totalwt+wt[k];
    }
    printf("\nprocess\tbt\ttat\twt\n\n\n");
    for(i=0;i<n;i++)
    {
        printf("\n%d\t%d\t%d\t%d",i+1,bt[i],tat[i],wt[i]);
    }
    printf("\n avg tat:%f",totaltat);
    printf("\n avg wt:%f",totalwt);
    return 0;
}
