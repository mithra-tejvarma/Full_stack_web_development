def binary_to_decimal(binary_str):
    """Convert an 8-bit binary string to a decimal integer."""
    return int(binary_str, 2)

def calculate_checksum(data_list):
    """Calculate the UDP checksum for given binary data strings."""
    total_sum = sum(binary_to_decimal(data) for data in data_list)

    # If sum exceeds 8 bits, wrap around
    while total_sum > 0xFF:
        total_sum = (total_sum & 0xFF) + (total_sum >> 8)

    # Take 1's complement
    checksum = ~total_sum & 0xFF

    # Convert to 8-bit binary string
    return format(checksum, '08b')

def verify_checksum(data_list, checksum):
    """Verify the UDP checksum."""
    total_sum = binary_to_decimal(checksum)  # Start with checksum

    # Add all binary data
    total_sum += sum(binary_to_decimal(data) for data in data_list)

    # If sum exceeds 8 bits, wrap around
    while total_sum > 0xFF:
        total_sum = (total_sum & 0xFF) + (total_sum >> 8)

    # Check if result is all 1s (255 in decimal or 0xFF)
    return total_sum == 0xFF

def factorial(n):
   if n==0:
      return 1
   else:
      result=1
      for i in range(1,n+1):
        result*=i
      return result

def is_strong(number):
    sum_of_factorials=0
    original_number=number
    while number>0:
       digit=number%10
       sum_of_factorials+=factorial(digit)
       number//=10
    return original_number==sum_of_factorials
