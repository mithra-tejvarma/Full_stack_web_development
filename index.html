FCFS (First-Come, First-Served) Scheduling Algorithm

The following C program implements the FCFS scheduling algorithm.

#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, bt[10], wt[10], tat[10], i, j;
    double awt = 0, atat = 0;

    printf("\nEnter the number of processes: ");
    scanf("%d", &n);

    printf("Enter process burst times:\n");
    for (i = 0; i < n; i++) {
        printf("P[%d]: ", i + 1);
        scanf("%d", &bt[i]);
    }

    wt[0] = 0;
    for (i = 1; i < n; i++) {
        wt[i] = wt[i - 1] + bt[i - 1];
    }

    for (i = 0; i < n; i++) {
        tat[i] = bt[i] + wt[i];
        awt += wt[i];
        atat += tat[i];
    }

    awt /= n;
    atat /= n;

    printf("\nPID\tBT\tWT\tTAT");
    for (i = 0; i < n; i++) {
        printf("\n%d\t%d\t%d\t%d", i + 1, bt[i], wt[i], tat[i]);
    }

    printf("\n\nAverage Waiting Time: %.2f", awt);
    printf("\nAverage Turnaround Time: %.2f\n", atat);

    return 0;
}
